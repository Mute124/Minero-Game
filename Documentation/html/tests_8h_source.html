87// function_view - adapted from here: https://vittorioromeo.info/index/blog/passing_functions_to_functions.html
88template <typename Func>
89class function_view;
90template <typename R, typename... P>
91class function_view<R(P...)> final
92{
93 private:
94 using func_type = R(P...);
95 using eraser_func_type = R(void*, P&&...);
96
97 mutable void* ptr_ = {};
98 mutable eraser_func_type* eraser = {};
99
100 public:
101 function_view() noexcept = default;
102
103 template <typename T>
104 function_view(T&& x) noexcept : ptr_{ reinterpret_cast<void*>(std::addressof(x)) }
105 {
106 eraser = [](void* ptr, P&&... xs) -> R
107 { return (*reinterpret_cast<std::add_pointer_t<std::remove_reference_t<T>>>(ptr))(std::forward<P>(xs)...); };
108 }
109
110 decltype(auto) operator()(P&&... xs) const
111 {
112 return eraser(ptr_, std::forward<P>(xs)...);
113 }
114
115 TOML_NODISCARD
116 operator bool() const noexcept
117 {
118 return !!ptr_;
119 }
120};

145template <typename T>
146inline bool parse_expected_value(std::string_view test_file,
147 uint32_t test_line,
148 std::string_view value_str,
149 const T& expected)
150{
151 INFO("["sv << test_file << ", line "sv << test_line << "] "sv
152 << "parse_expected_value(\""sv << value_str << "\")"sv)
153
154 std::string val;
155 static constexpr auto key = "val = "sv;
156 val.reserve(key.length() + value_str.length());
157 val.append(key);
158 val.append(value_str);
159
160 static constexpr auto is_val = [](char32_t codepoint)
161 {
162 if constexpr (impl::node_type_of<T> == node_type::string)
163 return codepoint == U'"' || codepoint == U'\'';
164 else
165 return !impl::is_whitespace(codepoint);
166 };
167
168 source_position pos{ 1, static_cast<source_index>(key.length()) };
169 source_position begin{}, end{};
170 {
171 impl::utf8_decoder decoder;
172 for (auto c : value_str)
173 {
174 decoder(static_cast<uint8_t>(c));
175 if (!decoder.has_code_point())
176 continue;
177
178 if (impl::is_ascii_vertical_whitespace(decoder.codepoint))
179 {
180 if (decoder.codepoint == U'\n')
181 {
182 pos.line++;
183 pos.column = source_index{ 1 };
184 }
185 continue;
186 }
187
188 pos.column++;
189 if (is_val(decoder.codepoint))
190 {
191 if (!begin)
192 begin = pos;
193 else
194 end = pos;
195 }
196 }
197 if (!end)
198 end = begin;
199 end.column++;
200 }
201
202 using value_type = impl::native_type_of<impl::remove_cvref<T>>;
203 value<value_type> val_parsed;
204 {
205 INFO("["sv << test_file << ", line "sv << test_line << "] "sv
206 << "parse_expected_value: Checking initial parse"sv)
207
208 bool stolen_value = false; // parsing_should_succeed invokes the functor more than once
209 const auto result = parsing_should_succeed(
210 test_file,
211 test_line,
212 std::string_view{ val },
213 [&](table&& tbl)
214 {
215 REQUIRE(tbl.size() == 1);
216 auto nv = tbl["val"sv];
217 REQUIRE(nv);
218 REQUIRE(nv.is<value_type>());
219 REQUIRE(nv.as<value_type>());
220 REQUIRE(nv.type() == impl::node_type_of<T>);
221 REQUIRE(nv.node());
222 REQUIRE(nv.node()->is<value_type>());
223 REQUIRE(nv.node()->as<value_type>());
224 REQUIRE(nv.node()->type() == impl::node_type_of<T>);
225
226 // check homogeneity
227 REQUIRE(nv.is_homogeneous());
228 REQUIRE(nv.is_homogeneous(node_type::none));
229 REQUIRE(nv.is_homogeneous(impl::node_type_of<T>));
230 REQUIRE(nv.is_homogeneous<value_type>());
231 REQUIRE(nv.node()->is_homogeneous());
232 REQUIRE(nv.node()->is_homogeneous(node_type::none));
233 REQUIRE(nv.node()->is_homogeneous(impl::node_type_of<T>));
234 REQUIRE(nv.node()->is_homogeneous<value_type>());
235 for (auto nt = impl::unwrap_enum(node_type::table); nt <= impl::unwrap_enum(node_type::date_time); nt++)
236 {
237 if (node_type{ nt } == impl::node_type_of<T>)
238 continue;
239 node* first_nonmatch{};
240 REQUIRE(!nv.is_homogeneous(node_type{ nt }));
241 REQUIRE(!nv.is_homogeneous(node_type{ nt }, first_nonmatch));
242 REQUIRE(first_nonmatch == nv.node());
243 REQUIRE(!nv.node()->is_homogeneous(node_type{ nt }));
244 REQUIRE(!nv.node()->is_homogeneous(node_type{ nt }, first_nonmatch));
245 REQUIRE(first_nonmatch == nv.node());
246 }
247
248 // check the raw value
249 REQUIRE(nv.node()->value<value_type>() == expected);
250 REQUIRE(nv.node()->value_or(T{}) == expected);
251 REQUIRE(nv.as<value_type>()->get() == expected);
252 REQUIRE(nv.value<value_type>() == expected);
253 REQUIRE(nv.value_or(T{}) == expected);
254 REQUIRE(nv.ref<value_type>() == expected);
255 REQUIRE(nv.node()->ref<value_type>() == expected);
256
257 // check the table relops
258 REQUIRE(tbl == table{ { { "val"sv, expected } } });
259 REQUIRE(!(tbl != table{ { { "val"sv, expected } } }));
260
261 // check value/node relops
262 CHECK_SYMMETRIC_EQUAL(*nv.as<value_type>(), *nv.as<value_type>());
263 CHECK_SYMMETRIC_EQUAL(*nv.as<value_type>(), expected);
264 CHECK_SYMMETRIC_EQUAL(nv, expected);
265
266 // make sure source info is correct
267 CHECK_SYMMETRIC_EQUAL(nv.node()->source().begin, begin);
268 CHECK_SYMMETRIC_EQUAL(nv.node()->source().end, end);
269
270 // check float identities etc
271 if constexpr (std::is_same_v<value_type, double>)
272 {
273 auto& float_node = *nv.as<value_type>();
274 const auto fpcls = impl::fpclassify(*float_node);
275 if (fpcls == impl::fp_class::nan)
276 {
277 CHECK_SYMMETRIC_EQUAL(float_node, std::numeric_limits<double>::quiet_NaN());
278 CHECK_SYMMETRIC_INEQUAL(float_node, std::numeric_limits<double>::infinity());
279 CHECK_SYMMETRIC_INEQUAL(float_node, -std::numeric_limits<double>::infinity());
280 CHECK_SYMMETRIC_INEQUAL(float_node, 1.0);
281 CHECK_SYMMETRIC_INEQUAL(float_node, 0.0);
282 CHECK_SYMMETRIC_INEQUAL(float_node, -1.0);
283 }
284 else if (fpcls == impl::fp_class::neg_inf || fpcls == impl::fp_class::pos_inf)
285 {
286 CHECK_SYMMETRIC_INEQUAL(float_node, std::numeric_limits<double>::quiet_NaN());
287 if (fpcls == impl::fp_class::neg_inf)
288 {
289 CHECK_SYMMETRIC_EQUAL(float_node, -std::numeric_limits<double>::infinity());
290 CHECK_SYMMETRIC_INEQUAL(float_node, std::numeric_limits<double>::infinity());
291 }
292 else
293 {
294 CHECK_SYMMETRIC_EQUAL(float_node, std::numeric_limits<double>::infinity());
295 CHECK_SYMMETRIC_INEQUAL(float_node, -std::numeric_limits<double>::infinity());
296 }
297 CHECK_SYMMETRIC_INEQUAL(float_node, 1.0);
298 CHECK_SYMMETRIC_INEQUAL(float_node, 0.0);
299 CHECK_SYMMETRIC_INEQUAL(float_node, -1.0);
300 }
301 else
302 {
303 CHECK_SYMMETRIC_INEQUAL(float_node, std::numeric_limits<double>::quiet_NaN());
304 CHECK_SYMMETRIC_INEQUAL(float_node, std::numeric_limits<double>::infinity());
305 CHECK_SYMMETRIC_INEQUAL(float_node, -std::numeric_limits<double>::infinity());
306 CHECK_SYMMETRIC_EQUAL(float_node, *float_node);
307 if (std::abs(*float_node) <= 1e10)
308 {
309 CHECK_SYMMETRIC_INEQUAL(float_node, *float_node + 100.0);
310 CHECK_SYMMETRIC_INEQUAL(float_node, *float_node - 100.0);
311 }
312 CHECK(float_node < std::numeric_limits<double>::infinity());
313 CHECK(float_node > -std::numeric_limits<double>::infinity());
314 }
315 }
316
317 // steal the val for round-trip tests
318 if (!stolen_value)
319 {
320 val_parsed = std::move(*nv.as<value_type>());
321 stolen_value = true;
322 }
323 });
324
325 if (!result)
326 return false;
327 }
328
329 // check round-tripping
330 {
331 INFO("["sv << test_file << ", line "sv << test_line << "] "sv
332 << "parse_expected_value: Checking round-trip"sv)
333 {
334 std::string str;
335 {
336 auto tbl = table{ { { "val"sv, *val_parsed } } };
337 std::ostringstream ss;
338 ss << tbl;
339 str = std::move(ss).str();
340 }
341
342 bool value_ok = true;
343 const auto parse_ok =
344 parsing_should_succeed(test_file,
345 test_line,
346 std::string_view{ str },
347 [&](table&& tbl)
348 {
349 REQUIRE(tbl.size() == 1);
350 auto nv = tbl["val"sv];
351 REQUIRE(nv);
352 REQUIRE(nv.as<value_type>());
353 REQUIRE(nv.node()->type() == impl::node_type_of<T>);
354
355 if (value_ok && nv.ref<value_type>() != expected)
356 {
357 value_ok = false;
358 FORCE_FAIL("Value was not the same after round-tripping"sv);
359 }
360 });
361
362 if (!parse_ok || value_ok)
363 return false;
364 }
365 }
366
367 return true;
368}
