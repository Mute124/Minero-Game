275 void GenerateSurface(const float* data)
276 {
277 // Clear the surface vertices and normals
278 surface_.clear();
279
280 // Iterate over each cell in the grid
281 for (int z = 0; z < length_ - 1; ++z)
282 {
283 for (int y = 0; y < height_ - 1; ++y)
284 {
285 for (int x = 0; x < width_ - 1; ++x)
286 {
287 // Retrieve the scalar values at the corners of the cell
288 float values[8] = {
289 data[Index(x, y, z)],
290 data[Index(x + 1, y, z)],
291 data[Index(x + 1, y, z + 1)],
292 data[Index(x, y, z + 1)],
293 data[Index(x, y + 1, z)],
294 data[Index(x + 1, y + 1, z)],
295 data[Index(x + 1, y + 1, z + 1)],
296 data[Index(x, y + 1, z + 1)]
297 };
298
299 // Determine the index into the edge table based on the scalar values
300 int index = 0;
301 if (values[0] < threshold_) index |= 1;
302 if (values[1] < threshold_) index |= 2;
303 if (values[2] < threshold_) index |= 4;
304 if (values[3] < threshold_) index |= 8;
305 if (values[4] < threshold_) index |= 16;
306 if (values[5] < threshold_) index |= 32;
307 if (values[6] < threshold_) index |= 64;
308 if (values[7] < threshold_) index |= 128;
309
310 // Use the edge table to determine the triangulation for this cell
311 int edgeFlags = edgeTable[index];
312 if (edgeFlags == 0) continue; // No triangles for this cell
313
314 // Interpolate the vertices and normals along the edges
315 if (edgeFlags & 1)
316 vertexList[0] = VertexInterpolate(x, y, z, values[0], values[1]);
317 if (edgeFlags & 2)
318 vertexList[1] = VertexInterpolate(x + 1, y, z, values[1], values[2]);
319 if (edgeFlags & 4)
320 vertexList[2] = VertexInterpolate(x + 1, y, z + 1, values[2], values[3]);
321 if (edgeFlags & 8)
322 vertexList[3] = VertexInterpolate(x, y, z + 1, values[3], values[0]);
323 if (edgeFlags & 16)
324 vertexList[4] = VertexInterpolate(x, y + 1, z, values[4], values[5]);
325 if (edgeFlags & 32)
326 vertexList[5] = VertexInterpolate(x + 1, y + 1, z, values[5], values[6]);
327 if (edgeFlags & 64)
328 vertexList[6] = VertexInterpolate(x + 1, y + 1, z + 1, values[6], values[7]);
329 if (edgeFlags & 128)
330 vertexList[7] = VertexInterpolate(x, y + 1, z + 1, values[7], values[4]);
331 if (edgeFlags & 256)
332 vertexList[8] = VertexInterpolate(x, y, z, values[0], values[4]);
333 if (edgeFlags & 512)
334 vertexList[9] = VertexInterpolate(x + 1, y, z, values[1], values[5]);
335 if (edgeFlags & 1024)
336 vertexList[10] = VertexInterpolate(x + 1, y, z + 1, values[2], values[6]);
337 if (edgeFlags & 2048)
338 vertexList[11] = VertexInterpolate(x, y, z + 1, values[3], values[7]);
339
340 // Create triangles from the vertices
341 for (int i = 0; triTable[index][i] != -1; i += 3)
342 {
343 int index1 = triTable[index][i];
344 int index2 = triTable[index][i + 1];
345 int index3 = triTable[index][i + 2];
346
347 // Calculate the surface normal using the cross product
348 Vector3 v1 = vertexList[index2] - vertexList[index1];
349 Vector3 v2 = vertexList[index3] - vertexList[index1];
350 Vector3 normal = Vector3::CrossProduct(v1, v2);
351
352 // Add the vertices and normals to the surface
353 surface_.push_back(vertexList[index1]);
354 surface_.push_back(normal);
355 surface_.push_back(vertexList[index2]);
356 surface_.push_back(normal);
357 surface_.push_back(vertexList[index3]);
358 surface_.push_back(normal);
359 }
360 }
