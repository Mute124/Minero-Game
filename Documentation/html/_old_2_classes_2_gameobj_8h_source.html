47 target.depth.format = 19; // DEPTH_COMPONENT_24BIT?
48 target.depth.mipmaps = 1;
49
50 // Attach color texture and depth texture to FBO
51 rlFramebufferAttach(target.id, target.texture.id, RL_ATTACHMENT_COLOR_CHANNEL0, RL_ATTACHMENT_TEXTURE2D, 0);
52 rlFramebufferAttach(target.id, target.depth.id, RL_ATTACHMENT_DEPTH, RL_ATTACHMENT_TEXTURE2D, 0);
53
54 // Check if fbo is complete with attachments (valid)
55 if (rlFramebufferComplete(target.id))
56 TRACELOG(LOG_INFO, "FBO: [ID %i] Framebuffer object created successfully", target.id);
57
58 rlDisableFramebuffer();
59 }
