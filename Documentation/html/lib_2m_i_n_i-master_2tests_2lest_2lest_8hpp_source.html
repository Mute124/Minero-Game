397class approx
398{
399public:
400 explicit approx ( double magnitude )
401 : epsilon_ { std::numeric_limits<float>::epsilon() * 100 }
402 , scale_ { 1.0 }
403 , magnitude_{ magnitude } {}
404
405 approx( approx const & other ) = default;
406
407 static approx custom() { return approx( 0 ); }
408
409 approx operator()( double new_magnitude )
410 {
411 approx appr( new_magnitude );
412 appr.epsilon( epsilon_ );
413 appr.scale ( scale_ );
414 return appr;
415 }
416
417 double magnitude() const { return magnitude_; }
418
419 approx & epsilon( double epsilon ) { epsilon_ = epsilon; return *this; }
420 approx & scale ( double scale ) { scale_ = scale; return *this; }
421
422 friend bool operator == ( double lhs, approx const & rhs )
423 {
424 // Thanks to Richard Harris for his help refining this formula.
425 return std::abs( lhs - rhs.magnitude_ ) < rhs.epsilon_ * ( rhs.scale_ + (std::min)( std::abs( lhs ), std::abs( rhs.magnitude_ ) ) );
426 }
427
428 friend bool operator == ( approx const & lhs, double rhs ) { return operator==( rhs, lhs ); }
429 friend bool operator != ( double lhs, approx const & rhs ) { return !operator==( lhs, rhs ); }
430 friend bool operator != ( approx const & lhs, double rhs ) { return !operator==( rhs, lhs ); }
431
432 friend bool operator <= ( double lhs, approx const & rhs ) { return lhs < rhs.magnitude_ || lhs == rhs; }
433 friend bool operator <= ( approx const & lhs, double rhs ) { return lhs.magnitude_ < rhs || lhs == rhs; }
434 friend bool operator >= ( double lhs, approx const & rhs ) { return lhs > rhs.magnitude_ || lhs == rhs; }
435 friend bool operator >= ( approx const & lhs, double rhs ) { return lhs.magnitude_ > rhs || lhs == rhs; }
436
437private:
438 double epsilon_;
439 double scale_;
440 double magnitude_;
441};

524inline std::string to_string( char chr )
525{
526 struct Tr { char chr; char const * str; } table[] =
527 {
528 {'\r', "'\\r'" }, {'\f', "'\\f'" },
529 {'\n', "'\\n'" }, {'\t', "'\\t'" },
530 };
531
532 for ( auto tr : table )
533 {
534 if ( chr == tr.chr )
535 return tr.str;
536 }
537
538 auto unprintable = [](char c){ return 0 <= c && c < ' '; };
539
540 return unprintable( chr )
541 ? to_string( static_cast<unsigned int>( chr ) )
542 : "\'" + std::string( 1, chr ) + "\'" ;

707inline
708auto make_memory_string( void const * item, std::size_t size ) -> std::string
709{
710 // reverse order for little endian architectures:
711
712 auto is_little_endian = []
713 {
714 union U { int i = 1; char c[ sizeof(int) ]; };
715
716 return 1 != U{}.c[ sizeof(int) - 1 ];
717 };
718
719 int i = 0, end = static_cast<int>( size ), inc = 1;
720
721 if ( is_little_endian() ) { i = end - 1; end = inc = -1; }
722
723 unsigned char const * bytes = static_cast<unsigned char const *>( item );
724
725 std::ostringstream os;
726 os << "0x" << std::setfill( '0' ) << std::hex;

1263 else if ( opt == "-l" || "--list-tests" == opt ) { option.list = true; continue; }
1264 else if ( opt == "-t" || "--time" == opt ) { option.time = true; continue; }
1265 else if ( opt == "-p" || "--pass" == opt ) { option.pass = true; continue; }
1266 else if ( opt == "-v" || "--verbose" == opt ) { option.verbose = true; continue; }
