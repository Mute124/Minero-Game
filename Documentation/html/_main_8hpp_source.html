207 void CreateSkybox()
208 {
209 // Load skybox model
210 // skybox creation.
211 Mesh cube = GenMeshCube(1.0f, 1.0f, 1.0f);
212 skybox = LoadModelFromMesh(cube);
213
214 bool useHDR = true;
215
216 // Load skybox shader and set required locations
217 // NOTE: Some locations are automatically set at shader loading
218 skybox.materials[0].shader =
219 LoadShader("resources/skybox.vs", "resources/skybox.fs");
220
221 // since the compiler complains about the references of such, these three vars
222 // are for the skybox shaders. They will be deleted after.
223 static int a = 7;
224 static int b = {useHDR ? 1 : 0};
225 static int c = 0;
226
227 SetShaderValue(
228 skybox.materials[0].shader,
229 GetShaderLocation(skybox.materials[0].shader, "environmentMap"), &a,
230 SHADER_UNIFORM_INT);
231 SetShaderValue(skybox.materials[0].shader,
232 GetShaderLocation(skybox.materials[0].shader, "doGamma"), &b,
233 SHADER_UNIFORM_INT);
234 SetShaderValue(skybox.materials[0].shader,
235 GetShaderLocation(skybox.materials[0].shader, "vflipped"), &b,
236 SHADER_UNIFORM_INT);
237
238 // Load cubemap shader and setup required shader locations
239 Shader shdrCubemap =
240 LoadShader("resources/cubemap.vs", "resources/cubemap.fs");
241
242 SetShaderValue(shdrCubemap,
243 GetShaderLocation(shdrCubemap, "equirectangularMap"), &c,
244 SHADER_UNIFORM_INT);
245
246 char skyboxFileName[256] = {0};
247
248 Texture2D panorama;
249
250 if (useHDR)
251 {
252 TextCopy(skyboxFileName, "resources/daytime.hdr");
253
254 // Load HDR panorama (sphere) texture
255 panorama = LoadTexture(skyboxFileName);
256
257 // Generate cubemap (texture with 6 quads-cube-mapping) from panorama HDR
258 // texture NOTE 1: New texture is generated rendering to texture, shader
259 // calculates the sphere->cube coordinates mapping NOTE 2: It seems on some
260 // Android devices WebGL, fbo does not properly support a FLOAT-based
261 // attachment, despite texture can be successfully created.. so using
262 // PIXELFORMAT_UNCOMPRESSED_R8G8B8A8 instead of
263 // PIXELFORMAT_UNCOMPRESSED_R32G32B32A32
264 skybox.materials[0].maps[MATERIAL_MAP_CUBEMAP].texture = Global::MeshTools::GenTextureCubemap(
265 shdrCubemap, panorama, 1024, PIXELFORMAT_UNCOMPRESSED_R8G8B8A8);
266
267 UnloadTexture(panorama); // Texture not required anymore, cubemap
268 // already generated
269 }
270 else
271 {
272 Image img = LoadImage("resources/skybox.png");
273 skybox.materials[0].maps[MATERIAL_MAP_CUBEMAP].texture = LoadTextureCubemap(
274 img, CUBEMAP_LAYOUT_AUTO_DETECT); // CUBEMAP_LAYOUT_PANORAMA
275 UnloadImage(img);
276 }
