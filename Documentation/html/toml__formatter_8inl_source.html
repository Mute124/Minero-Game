27 size_t toml_formatter_count_inline_columns(const node& node, size_t line_wrap_cols) noexcept
28 {
29 switch (node.type())
30 {
31 case node_type::table:
32 {
33 auto& tbl = *reinterpret_cast<const table*>(&node);
34 if (tbl.empty())
35 return 2u; // "{}"
36 size_t weight = 3u; // "{ }"
37 for (auto&& [k, v] : tbl)
38 {
39 weight += k.length() + toml_formatter_count_inline_columns(v, line_wrap_cols) + 2u; // + ", "
40 if (weight >= line_wrap_cols)
41 break;
42 }
43 return weight;
44 }
45
46 case node_type::array:
47 {
48 auto& arr = *reinterpret_cast<const array*>(&node);
49 if (arr.empty())
50 return 2u; // "[]"
51 size_t weight = 3u; // "[ ]"
52 for (auto& elem : arr)
53 {
54 weight += toml_formatter_count_inline_columns(elem, line_wrap_cols) + 2u; // + ", "
55 if (weight >= line_wrap_cols)
56 break;
57 }
58 return weight;
59 }
60
61 case node_type::string:
62 {
63 // todo: proper utf8 decoding?
64 // todo: tab awareness?
65 auto& str = (*reinterpret_cast<const value<std::string>*>(&node)).get();
66 return str.length() + 2u; // + ""
67 }
68
69 case node_type::integer:
70 {
71 auto val = (*reinterpret_cast<const value<int64_t>*>(&node)).get();
72 if (!val)
73 return 1u;
74 size_t weight = {};
75 if (val < 0)
76 {
77 weight += 1u;
78 val *= -1;
79 }
80 return weight + static_cast<size_t>(log10(static_cast<double>(val))) + 1u;
81 }
82
83 case node_type::floating_point:
84 {
85 auto val = (*reinterpret_cast<const value<double>*>(&node)).get();
86 if (val == 0.0)
87 return 3u; // "0.0"
88 size_t weight = 2u; // ".0"
89 if (val < 0.0)
90 {
91 weight += 1u;
92 val *= -1.0;
93 }
94 return weight + static_cast<size_t>(log10(val)) + 1u;
95 break;
96 }
97
98 case node_type::boolean: return 5u;
99 case node_type::date: [[fallthrough]];
100 case node_type::time: return 10u;
101 case node_type::date_time: return 30u;
102 case node_type::none: TOML_UNREACHABLE;
103 default: TOML_UNREACHABLE;
104 }
105
106 TOML_UNREACHABLE;
107 }

137 void toml_formatter::print_inline(const table& tbl)
138 {
139 if (tbl.empty())
140 {
141 print_unformatted("{}"sv);
142 return;
143 }
144
145 print_unformatted("{ "sv);
146
147 bool first = false;
148 for (auto&& [k, v] : tbl)
149 {
150 if (first)
151 print_unformatted(", "sv);
152 first = true;
153
154 print(k);
155 if (terse_kvps())
156 print_unformatted("="sv);
157 else
158 print_unformatted(" = "sv);
159
160 const auto type = v.type();
161 TOML_ASSUME(type != node_type::none);
162 switch (type)
163 {
164 case node_type::table: print_inline(*reinterpret_cast<const table*>(&v)); break;
165 case node_type::array: print(*reinterpret_cast<const array*>(&v)); break;
166 default: print_value(v, type);
167 }
168 }
169
170 print_unformatted(" }"sv);
171 }

173 TOML_EXTERNAL_LINKAGE
174 void toml_formatter::print(const array& arr)
175 {
176 if (arr.empty())
177 {
178 print_unformatted("[]"sv);
179 return;
180 }
181
182 const auto original_indent = indent();
183 const auto multiline = TOML_ANON_NAMESPACE::toml_formatter_forces_multiline(
184 arr,
185 120u,
186 indent_columns() * static_cast<size_t>(original_indent < 0 ? 0 : original_indent));
187
188 print_unformatted("["sv);
189
190 if (multiline)
191 {
192 if (original_indent < 0)
193 indent(0);
194 if (indent_array_elements())
195 increase_indent();
196 }
197 else
198 print_unformatted(' ');
199
200 for (size_t i = 0; i < arr.size(); i++)
201 {
202 if (i > 0u)
203 {
204 print_unformatted(',');
205 if (!multiline)
206 print_unformatted(' ');
207 }
208
209 if (multiline)
210 {
211 print_newline(true);
212 print_indent();
213 }
214
215 auto& v = arr[i];
216 const auto type = v.type();
217 TOML_ASSUME(type != node_type::none);
218 switch (type)
219 {
220 case node_type::table: print_inline(*reinterpret_cast<const table*>(&v)); break;
221 case node_type::array: print(*reinterpret_cast<const array*>(&v)); break;
222 default: print_value(v, type);
223 }
224 }
225 if (multiline)
226 {
227 indent(original_indent);
228 print_newline(true);
229 print_indent();
230 }
231 else
232 print_unformatted(' ');
233
234 print_unformatted("]"sv);
235 }

237 TOML_EXTERNAL_LINKAGE
238 void toml_formatter::print(const table& tbl)
239 {
240 static constexpr auto is_non_inline_array_of_tables = [](const node& n) noexcept
241 {
242 const auto arr = n.as_array();
243 if (!arr || !arr->is_array_of_tables())
244 return false;
245
246 return !reinterpret_cast<const table*>(&(*arr)[0])->is_inline();
247 };
248
249 // values, arrays, and inline tables/table arrays
250 for (auto&& [k, v] : tbl)
251 {
252 const auto type = v.type();
253 if ((type == node_type::table && !reinterpret_cast<const table*>(&v)->is_inline())
254 || (type == node_type::array && is_non_inline_array_of_tables(v)))
255 continue;
256
257 pending_table_separator_ = true;
258 print_newline();
259 print_indent();
260 print(k);
261 if (terse_kvps())
262 print_unformatted("="sv);
263 else
264 print_unformatted(" = "sv);
265 TOML_ASSUME(type != node_type::none);
266 switch (type)
267 {
268 case node_type::table: print_inline(*reinterpret_cast<const table*>(&v)); break;
269 case node_type::array: print(*reinterpret_cast<const array*>(&v)); break;
270 default: print_value(v, type);
271 }
272 }
273
274 const auto print_key_path = [&]()
275 {
276 size_t i{};
277 for (const auto k : key_path_)
278 {
279 if (i++)
280 print_unformatted('.');
281 print(*k);
282 }
283 };
284
285 // non-inline tables
286 for (auto&& [k, v] : tbl)
287 {
288 const auto type = v.type();
289 if (type != node_type::table || reinterpret_cast<const table*>(&v)->is_inline())
290 continue;
291 auto& child_tbl = *reinterpret_cast<const table*>(&v);
292
293 // we can skip indenting and emitting the headers for tables that only contain other tables
294 // (so we don't over-nest)
295 size_t child_value_count{}; // includes inline tables and non-table arrays
296 size_t child_table_count{};
297 size_t child_table_array_count{};
298 for (auto&& [child_k, child_v] : child_tbl)
299 {
300 TOML_UNUSED(child_k);
301 const auto child_type = child_v.type();
302 TOML_ASSUME(child_type != node_type::none);
303 switch (child_type)
304 {
305 case node_type::table:
306 if (reinterpret_cast<const table*>(&child_v)->is_inline())
307 child_value_count++;
308 else
309 child_table_count++;
310 break;
311
312 case node_type::array:
313 if (is_non_inline_array_of_tables(child_v))
314 child_table_array_count++;
315 else
316 child_value_count++;
317 break;
318
319 default: child_value_count++;
320 }
321 }
322 bool skip_self = false;
323 if (child_value_count == 0u && (child_table_count > 0u || child_table_array_count > 0u))
324 skip_self = true;
325
326 key_path_.push_back(&k);
327
328 if (!skip_self)
329 {
330 print_pending_table_separator();
331 if (indent_sub_tables())
332 increase_indent();
333 print_indent();
334 print_unformatted("["sv);
335 print_key_path();
336 print_unformatted("]"sv);
337 pending_table_separator_ = true;
338 }
339
340 print(child_tbl);
341
342 key_path_.pop_back();
343 if (!skip_self && indent_sub_tables())
344 decrease_indent();
345 }
346
347 // table arrays
348 for (auto&& [k, v] : tbl)
349 {
350 if (!is_non_inline_array_of_tables(v))
351 continue;
352 auto& arr = *reinterpret_cast<const array*>(&v);
353
354 if (indent_sub_tables())
355 increase_indent();
356 key_path_.push_back(&k);
357
358 for (size_t i = 0; i < arr.size(); i++)
359 {
360 print_pending_table_separator();
361 print_indent();
362 print_unformatted("[["sv);
363 print_key_path();
364 print_unformatted("]]"sv);
365 pending_table_separator_ = true;
366 print(*reinterpret_cast<const table*>(&arr[i]));
367 }
368
369 key_path_.pop_back();
370 if (indent_sub_tables())
371 decrease_indent();
372 }
373 }

375 TOML_EXTERNAL_LINKAGE
376 void toml_formatter::print()
377 {
378 if (dump_failed_parse_result())
379 return;
380
381 switch (auto source_type = source().type())
382 {
383 case node_type::table:
384 {
385 auto& tbl = *reinterpret_cast<const table*>(&source());
386 if (tbl.is_inline())
387 print_inline(tbl);
388 else
389 {
390 decrease_indent(); // so root kvps and tables have the same indent
391 print(tbl);
392 }
393 break;
394 }
395
396 case node_type::array: print(*reinterpret_cast<const array*>(&source())); break;
397
398 default: print_value(source(), source_type);
399 }
400 }
