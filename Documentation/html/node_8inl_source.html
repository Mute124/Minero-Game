120 bool TOML_CALLCONV node_deep_equality(const node* lhs, const node* rhs) noexcept
121 {
122 // both same or both null
123 if (lhs == rhs)
124 return true;
125
126 // lhs null != rhs null or different types
127 if ((!lhs != !rhs) || lhs->type() != rhs->type())
128 return false;
129
130 return lhs->visit(
131 [=](auto& l) noexcept
132 {
133 using concrete_type = remove_cvref<decltype(l)>;
134
135 return l == *(rhs->as<concrete_type>());
136 });
137 }
138}
