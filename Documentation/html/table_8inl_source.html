239 TOML_EXTERNAL_LINKAGE
240 table& table::prune(bool recursive)& noexcept
241 {
242 if (map_.empty())
243 return *this;
244
245 for (auto it = map_.begin(); it != map_.end();)
246 {
247 if (auto arr = it->second->as_array())
248 {
249 if (recursive)
250 arr->prune(true);
251
252 if (arr->empty())
253 {
254 it = map_.erase(it);
255 continue;
256 }
257 }
258 else if (auto tbl = it->second->as_table())
259 {
260 if (recursive)
261 tbl->prune(true);
262
263 if (tbl->empty())
264 {
265 it = map_.erase(it);
266 continue;
267 }
268 }
269 it++;
270 }
271
272 return *this;
273 }

288 TOML_EXTERNAL_LINKAGE
289 bool TOML_CALLCONV table::equal(const table& lhs, const table& rhs) noexcept
290 {
291 if (&lhs == &rhs)
292 return true;
293 if (lhs.map_.size() != rhs.map_.size())
294 return false;
295
296 for (auto l = lhs.map_.begin(), r = rhs.map_.begin(), e = lhs.map_.end(); l != e; l++, r++)
297 {
298 if (l->first != r->first)
299 return false;
300
301 const auto lhs_type = l->second->type();
302 const node& rhs_ = *r->second;
303 const auto rhs_type = rhs_.type();
304 if (lhs_type != rhs_type)
305 return false;
306
307 const bool equal = l->second->visit(
308 [&](const auto& lhs_) noexcept
309 { return lhs_ == *reinterpret_cast<std::remove_reference_t<decltype(lhs_)>*>(&rhs_); });
310 if (!equal)
311 return false;
312 }
313 return true;
314 }
