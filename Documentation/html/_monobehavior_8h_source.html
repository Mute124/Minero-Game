277 {
278 TextureCubemap cubemap = {0};
279
280 rlDisableBackfaceCulling(); // Disable backface culling to render inside the
281 // cube
282
283 // STEP 1: Setup framebuffer
284 //------------------------------------------------------------------------------------------
285 unsigned int rbo = rlLoadTextureDepth(size, size, true);
286 cubemap.id = rlLoadTextureCubemap(0, size, format);
287
288 unsigned int fbo = rlLoadFramebuffer(size, size);
289 rlFramebufferAttach(fbo, rbo, RL_ATTACHMENT_DEPTH, RL_ATTACHMENT_RENDERBUFFER,
290 0);
291 rlFramebufferAttach(fbo, cubemap.id, RL_ATTACHMENT_COLOR_CHANNEL0,
292 RL_ATTACHMENT_CUBEMAP_POSITIVE_X, 0);
293
294 // Check if framebuffer is complete with attachments (valid)
295 if (rlFramebufferComplete(fbo))
296 TraceLog(LOG_INFO, "FBO: [ID %i] Framebuffer object created successfully",
297 fbo);
298 //------------------------------------------------------------------------------------------
299
300 // STEP 2: Draw to framebuffer
301 //------------------------------------------------------------------------------------------
302 // NOTE: Shader is used to convert HDR equirectangular environment map to
303 // cubemap equivalent (6 faces)
304 rlEnableShader(shader.id);
305
306 // Define projection matrix and send it to shader
307 Matrix matFboProjection = MatrixPerspective(
308 90.0 * DEG2RAD, 1.0, RL_CULL_DISTANCE_NEAR, RL_CULL_DISTANCE_FAR);
309 rlSetUniformMatrix(shader.locs[SHADER_LOC_MATRIX_PROJECTION],
310 matFboProjection);
311
312 // Define view matrix for every side of the cubemap
313 Matrix fboViews[6] = {
314 MatrixLookAt(Vector3{0.0f, 0.0f, 0.0f}, Vector3{1.0f, 0.0f, 0.0f},
315 Vector3{0.0f, -1.0f, 0.0f}),
316 MatrixLookAt(Vector3{0.0f, 0.0f, 0.0f}, Vector3{-1.0f, 0.0f, 0.0f},
317 Vector3{0.0f, -1.0f, 0.0f}),
318 MatrixLookAt(Vector3{0.0f, 0.0f, 0.0f}, Vector3{0.0f, 1.0f, 0.0f},
319 Vector3{0.0f, 0.0f, 1.0f}),
320 MatrixLookAt(Vector3{0.0f, 0.0f, 0.0f}, Vector3{0.0f, -1.0f, 0.0f},
321 Vector3{0.0f, 0.0f, -1.0f}),
322 MatrixLookAt(Vector3{0.0f, 0.0f, 0.0f}, Vector3{0.0f, 0.0f, 1.0f},
323 Vector3{0.0f, -1.0f, 0.0f}),
324 MatrixLookAt(Vector3{0.0f, 0.0f, 0.0f}, Vector3{0.0f, 0.0f, -1.0f},
325 Vector3{0.0f, -1.0f, 0.0f})};
326
327 rlViewport(0, 0, size, size); // Set viewport to current fbo dimensions
328
329 // Activate and enable texture for drawing to cubemap faces
330 rlActiveTextureSlot(0);
331 rlEnableTexture(panorama.id);
332
333 for (int i = 0; i < 6; i++)
334 {
335 // Set the view matrix for the current cube face
336 rlSetUniformMatrix(shader.locs[SHADER_LOC_MATRIX_VIEW], fboViews[i]);
337
338 // Select the current cubemap face attachment for the fbo
339 // WARNING: This function by default enables->attach->disables fbo!!!
340 rlFramebufferAttach(fbo, cubemap.id, RL_ATTACHMENT_COLOR_CHANNEL0,
341 RL_ATTACHMENT_CUBEMAP_POSITIVE_X + i, 0);
342 rlEnableFramebuffer(fbo);
343
344 // Load and draw a cube, it uses the current enabled texture
345 rlClearScreenBuffers();
346 rlLoadDrawCube();
347
348 // ALTERNATIVE: Try to use internal batch system to draw the cube instead of
349 // rlLoadDrawCube for some reason this method does not work, maybe due to
350 // cube triangles definition? normals pointing out?
351 // TODO: Investigate this issue...
352 // rlSetTexture(panorama.id); // WARNING: It must be called after enabling
353 // current framebuffer if using internal batch system!
354 // rlClearScreenBuffers();
355 // DrawCubeV(Vector3Zero(), Vector3One(), WHITE);
356 // rlDrawRenderBatchActive();
357 }
358 //------------------------------------------------------------------------------------------
359
360 // STEP 3: Unload framebuffer and reset state
361 //------------------------------------------------------------------------------------------
362 rlDisableShader(); // Unbind shader
363 rlDisableTexture(); // Unbind texture
364 rlDisableFramebuffer(); // Unbind framebuffer
365 rlUnloadFramebuffer(fbo); // Unload framebuffer (and automatically attached
366 // depth texture/renderbuffer)
367
368 // Reset viewport dimensions to default
369 rlViewport(0, 0, rlGetFramebufferWidth(), rlGetFramebufferHeight());
370 rlEnableBackfaceCulling();
371 //------------------------------------------------------------------------------------------
372
373 cubemap.width = size;
374 cubemap.height = size;
375 cubemap.mipmaps = 1;
376 cubemap.format = format;
377
378 return cubemap;

604 static bool CheckCollisionRayBox(Ray ray, BoundingBox box, float *outDistance)
605 {
606 // Compute the inverse of the ray direction
607 Vector3 invDir = {1.0f / ray.direction.x, 1.0f / ray.direction.y,
608 1.0f / ray.direction.z};
609
610 // Compute the sign of the inverse direction
611 int signX = (invDir.x < 0.0f) ? 1 : 0;
612 int signY = (invDir.y < 0.0f) ? 1 : 0;
613 int signZ = (invDir.z < 0.0f) ? 1 : 0;
614
615 // Compute the min and max values for each axis
616 float tmin = (box.min.x - ray.position.x) * invDir.x;
617 float tmax = (box.max.x - ray.position.x) * invDir.x;
618 float tymin = (box.min.y - ray.position.y) * invDir.y;
619 float tymax = (box.max.y - ray.position.y) * invDir.y;
620
621 // Swap the min and max values if necessary
622 if (tmin > tmax)
623 {
624 float temp = tmin;
625 tmin = tmax;
626 tmax = temp;
627 }
628
629 if (tymin > tymax)
630 {
631 float temp = tymin;
632 tymin = tymax;
633 tymax = temp;
634 }
635
636 // Check for a miss
637 if ((tmin > tymax) || (tymin > tmax))
638 {
639 return false;
640 }
641
642 // Compute the final min and max values for each axis
643 if (tymin > tmin)
644 {
645 tmin = tymin;
646 }
647
648 if (tymax < tmax)
649 {
650 tmax = tymax;
651 }
652
653 float tzmin = (box.min.z - ray.position.z) * invDir.z;
654 float tzmax = (box.max.z - ray.position.z) * invDir.z;
655
656 // Swap the min and max values if necessary
657 if (tzmin > tzmax)
658 {
659 float temp = tzmin;
660 tzmin = tzmax;
661 tzmax = temp;
662 }
663
664 // Check for a miss
665 if ((tmin > tzmax) || (tzmin > tmax))
666 {
667 return false;
668 }
669
670 // Compute the final min and max values for each axis
671 if (tzmin > tmin)
672 {
673 tmin = tzmin;
674 }
675
676 if (tzmax < tmax)
677 {
678 tmax = tzmax;
679 }
680
681 // Store the distance to the box in the output variable
682 if (outDistance)
683 {
684 *outDistance = tmin;
685 }
686
687 // The ray intersects the box
688 return true;

690 static Vector3 CameraToWorld(Camera3D camera)
691 {
692 float distance;
693 // Calculate the ray from the camera to the mouse position
694 Vector2 mousePos = GetMousePosition();
695 Vector3 mousePos3 = {mousePos.x, mousePos.y, 0};
696 // Get the direction vector of the ray
697 Vector3 rayDirection =
698 Vector3Normalize(Vector3Subtract(mousePos3, camera.position));
699
700 // Multiply the direction vector with a scalar value to increase the length of
701 // the ray
702 float rayLength = 40.0f;
703 rayDirection = Vector3Scale(rayDirection, rayLength);
704
705 // Create the ray with the modified direction
706 Ray ray = {camera.position, rayDirection};
707
708 // Calculate the distance from the camera to the ground plane
709 distance = -(camera.position.y / ray.position.y);
710
711 // Calculate the point where the ray intersects the ground plane
712 Vector3 groundPoint =
713 Vector3Add(camera.position, Vector3Scale(ray.position, distance));
714
715 return groundPoint;
