35 TOML_EXTERNAL_LINKAGE
36 array::array(const impl::array_init_elem* b, const impl::array_init_elem* e)
37 {
38#if TOML_LIFETIME_HOOKS
39 TOML_ARRAY_CREATED;
40#endif
41
42 TOML_ASSERT_ASSUME(b);
43 TOML_ASSERT_ASSUME(e);
44 TOML_ASSERT_ASSUME(b <= e);
45
46 if TOML_UNLIKELY(b == e)
47 return;
48
49 size_t cap{};
50 for (auto it = b; it != e; it++)
51 {
52 if (it->value)
53 cap++;
54 }
55 if TOML_UNLIKELY(!cap)
56 return;
57
58 elems_.reserve(cap);
59 for (; b != e; b++)
60 {
61 if (b->value)
62 elems_.push_back(std::move(b->value));
63 }
64 }

114 TOML_EXTERNAL_LINKAGE
115 void array::preinsertion_resize(size_t idx, size_t count)
116 {
117 TOML_ASSERT(idx <= elems_.size());
118 TOML_ASSERT_ASSUME(count >= 1u);
119 const auto old_size = elems_.size();
120 const auto new_size = old_size + count;
121 const auto inserting_at_end = idx == old_size;
122 elems_.resize(new_size);
123 if (!inserting_at_end)
124 {
125 for (size_t left = old_size, right = new_size - 1u; left-- > idx; right--)
126 elems_[right] = std::move(elems_[left]);
127 }
128 }

253 {

268 TOML_EXTERNAL_LINKAGE
269 array& array::flatten()&
270 {
271 if (elems_.empty())
272 return *this;
273
274 bool requires_flattening = false;
275 size_t size_after_flattening = elems_.size();
276 for (size_t i = elems_.size(); i-- > 0u;)
277 {
278 auto arr = elems_[i]->as_array();
279 if (!arr)
280 continue;
281 size_after_flattening--; // discount the array itself
282 const auto leaf_count = arr->total_leaf_count();
283 if (leaf_count > 0u)
284 {
285 requires_flattening = true;
286 size_after_flattening += leaf_count;
287 }
288 else
289 elems_.erase(elems_.cbegin() + static_cast<ptrdiff_t>(i));
290 }
291
292 if (!requires_flattening)
293 return *this;
294
295 elems_.reserve(size_after_flattening);
296
297 size_t i = 0;
298 while (i < elems_.size())
299 {
300 auto arr = elems_[i]->as_array();
301 if (!arr)
302 {
303 i++;
304 continue;
305 }
306
307 impl::node_ptr arr_storage = std::move(elems_[i]);
308 const auto leaf_count = arr->total_leaf_count();
309 if (leaf_count > 1u)
310 preinsertion_resize(i + 1u, leaf_count - 1u);
311 flatten_child(std::move(*arr), i); // increments i
312 }
313
314 return *this;
315 }

317 TOML_EXTERNAL_LINKAGE
318 array& array::prune(bool recursive)& noexcept
319 {
320 if (elems_.empty())
321 return *this;
322
323 for (size_t i = elems_.size(); i-- > 0u;)
324 {
325 if (auto arr = elems_[i]->as_array())
326 {
327 if (recursive)
328 arr->prune(true);
329 if (arr->empty())
330 elems_.erase(elems_.cbegin() + static_cast<ptrdiff_t>(i));
331 }
332 else if (auto tbl = elems_[i]->as_table())
333 {
334 if (recursive)
335 tbl->prune(true);
336 if (tbl->empty())
337 elems_.erase(elems_.cbegin() + static_cast<ptrdiff_t>(i));
338 }
339 }
340
341 return *this;
342 }

356 TOML_EXTERNAL_LINKAGE
357 bool TOML_CALLCONV array::equal(const array& lhs, const array& rhs) noexcept
358 {
359 if (&lhs == &rhs)
360 return true;
361 if (lhs.elems_.size() != rhs.elems_.size())
362 return false;
363 for (size_t i = 0, e = lhs.elems_.size(); i < e; i++)
364 {
365 const auto lhs_type = lhs.elems_[i]->type();
366 const node& rhs_ = *rhs.elems_[i];
367 const auto rhs_type = rhs_.type();
368 if (lhs_type != rhs_type)
369 return false;
370
371 const bool equal = lhs.elems_[i]->visit(
372 [&](const auto& lhs_) noexcept
373 { return lhs_ == *reinterpret_cast<std::remove_reference_t<decltype(lhs_)>*>(&rhs_); });
374 if (!equal)
375 return false;
376 }
377 return true;
378 }
