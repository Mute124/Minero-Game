121// Create a light and get shader locations
122Light CreateLight(int type, Vector3 position, Vector3 target, Color color, std::vector<Shader> shaders)
123{
124 Light light = { 0 };
125
126 if (lightsCount < MAX_LIGHTS)
127 {
128 light.enabled = true;
129 light.type = type;
130 light.position = position;
131 light.target = target;
132 light.color = color;
133
134 // TODO: Below code doesn't look good to me,
135 // it assumes a specific shader naming and structure
136 // Probably this implementation could be improved
137 char enabledName[32] = "lights[x].enabled\0";
138 char typeName[32] = "lights[x].type\0";
139 char posName[32] = "lights[x].position\0";
140 char targetName[32] = "lights[x].target\0";
141 char colorName[32] = "lights[x].color\0";
142
143 // Set location name [x] depending on lights count
144 enabledName[7] = '0' + lightsCount;
145 typeName[7] = '0' + lightsCount;
146 posName[7] = '0' + lightsCount;
147 targetName[7] = '0' + lightsCount;
148 colorName[7] = '0' + lightsCount;
149
150 for (size_t i = 0; i < shaders.size(); i++)
151 {
152 light.enabledLoc.push_back(GetShaderLocation(shaders[i], enabledName));
153 light.typeLoc.push_back(GetShaderLocation(shaders[i], typeName));
154 light.posLoc.push_back(GetShaderLocation(shaders[i], posName));
155 light.targetLoc.push_back(GetShaderLocation(shaders[i], targetName));
156 light.colorLoc.push_back(GetShaderLocation(shaders[i], colorName));
157 }
158
159 light.shaders = shaders;
160
161 lightsCount++;
162 }
163
164 return light;
165}

167// Send light properties to shader
168// NOTE: Light shader locations should be available
169void UpdateLightValues(Light light)
170{
171 for (size_t i = 0; i < light.shaders.size(); i++)
172 {
173 // Send to shader light enabled state and type
174 SetShaderValue(light.shaders[i], light.enabledLoc[i], &light.enabled, SHADER_UNIFORM_INT);
175 SetShaderValue(light.shaders[i], light.typeLoc[i], &light.type, SHADER_UNIFORM_INT);
176
177 // Send to shader light position values
178 float position[3] = { light.position.x, light.position.y, light.position.z };
179 SetShaderValue(light.shaders[i], light.posLoc[i], position, SHADER_UNIFORM_VEC3);
180
181 // Send to shader light target position values
182 float target[3] = { light.target.x, light.target.y, light.target.z };
183 SetShaderValue(light.shaders[i], light.targetLoc[i], target, SHADER_UNIFORM_VEC3);
184
185 // Send to shader light color values
186 float color[4] = { (float)light.color.r / (float)255, (float)light.color.g / (float)255,
187 (float)light.color.b / (float)255, (float)light.color.a / (float)255 };
188 SetShaderValue(light.shaders[i], light.colorLoc[i], color, SHADER_UNIFORM_VEC4);
189 }
190}
