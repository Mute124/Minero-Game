26 enum class TOML_CLOSED_FLAGS_ENUM formatted_string_traits : unsigned
27 {
28 none,
29 line_breaks = 1u << 0, // \n
30 tabs = 1u << 1, // \t
31 control_chars = 1u << 2, // also includes non-ascii vertical whitespace
32 single_quotes = 1u << 3,
33 non_bare = 1u << 4, // anything not satisfying "is bare key character"
34 non_ascii = 1u << 5, // any codepoint >= 128
35
36 all = (non_ascii << 1u) - 1u
37 };

116 void formatter::print_string(std::string_view str, bool allow_multi_line, bool allow_bare)
117 {
118 if (str.empty())
119 {
120 print_unformatted(literal_strings_allowed() ? "''"sv : "\"\""sv);
121 return;
122 }
123
124 // pre-scan the string to determine how we should output it
125 formatted_string_traits traits = {};
126
127 if (!allow_bare)
128 traits |= formatted_string_traits::non_bare;
129 bool unicode_allowed = unicode_strings_allowed();
130
131 // ascii fast path
132 if (is_ascii(str.data(), str.length()))
133 {
134 for (auto c : str)
135 {
136 switch (c)
137 {
138 case '\n': traits |= formatted_string_traits::line_breaks; break;
139 case '\t': traits |= formatted_string_traits::tabs; break;
140 case '\'': traits |= formatted_string_traits::single_quotes; break;
141 default:
142 {
143 if TOML_UNLIKELY(is_control_character(c))
144 traits |= formatted_string_traits::control_chars;
145
146 if (!is_ascii_bare_key_character(static_cast<char32_t>(c)))
147 traits |= formatted_string_traits::non_bare;
148 break;
149 }
150 }
151
152 static constexpr auto all_ascii_traits =
153 formatted_string_traits::all & ~formatted_string_traits::non_ascii;
154 if (traits == all_ascii_traits)
155 break;
156 }
157 }
158
159 // unicode slow path
160 else
161 {
162 traits |= formatted_string_traits::non_ascii;
163 utf8_decoder decoder;
164
165 // if the unicode is malformed just treat the string as a single-line non-literal and
166 // escape all non-ascii characters (to ensure round-tripping and help with diagnostics)
167 const auto bad_unicode = [&]() noexcept
168 {
169 traits &= ~formatted_string_traits::line_breaks;
170 traits |= formatted_string_traits::control_chars | formatted_string_traits::non_bare;
171 unicode_allowed = false;
172 };
173
174 for (auto c : str)
175 {
176 decoder(c);
177
178 if TOML_UNLIKELY(decoder.error())
179 {
180 bad_unicode();
181 break;
182 }
183
184 if (!decoder.has_code_point())
185 continue;
186
187 switch (decoder.codepoint)
188 {
189 case U'\n': traits |= formatted_string_traits::line_breaks; break;
190 case U'\t': traits |= formatted_string_traits::tabs; break;
191 case U'\'': traits |= formatted_string_traits::single_quotes; break;
192 default:
193 {
194 if TOML_UNLIKELY(is_control_character(decoder.codepoint)
195 || is_non_ascii_vertical_whitespace(decoder.codepoint))
196 traits |= formatted_string_traits::control_chars;
197
198 if (!is_bare_key_character(decoder.codepoint))
199 traits |= formatted_string_traits::non_bare;
200 break;
201 }
202 }
203 }
204
205 if (decoder.needs_more_input())
206 bad_unicode();
207 }
208
209 // if the string meets the requirements of being 'bare' we can emit a bare string
210 // (bare strings are composed of letters and numbers; no whitespace, control chars, quotes, etc)
211 if (!(traits & formatted_string_traits::non_bare)
212 && (!(traits & formatted_string_traits::non_ascii) || unicode_allowed))
213 {
214 print_unformatted(str);
215 return;
216 }
217
218 // determine if this should be a multi-line string (triple-quotes)
219 const auto multi_line = allow_multi_line //
220 && multi_line_strings_allowed() //
221 && !!(traits & formatted_string_traits::line_breaks);
222
223 // determine if this should be a literal string (single-quotes with no escaping)
224 const auto literal = literal_strings_allowed() //
225 && !(traits & formatted_string_traits::control_chars) //
226 && (!(traits & formatted_string_traits::single_quotes) || multi_line) //
227 && (!(traits & formatted_string_traits::tabs) || real_tabs_in_strings_allowed()) //
228 && (!(traits & formatted_string_traits::non_ascii) || unicode_allowed);
229
230 // literal strings (single quotes, no escape codes)
231 if (literal)
232 {
233 const auto quot = multi_line ? R"(''')"sv : R"(')"sv;
234 print_unformatted(quot);
235 print_unformatted(str);
236 print_unformatted(quot);
237 return;
238 }
239
240 // anything from here down is a non-literal string, so requires iteration and escaping.
241 print_unformatted(multi_line ? R"(""")"sv : R"(")"sv);
242
243 const auto real_tabs_allowed = real_tabs_in_strings_allowed();
244
245 // ascii fast path
246 if (!(traits & formatted_string_traits::non_ascii))
247 {
248 for (auto c : str)
249 {
250 switch (c)
251 {
252 case '"': print_to_stream(*stream_, R"(\")"sv); break;
253 case '\\': print_to_stream(*stream_, R"(\\)"sv); break;
254 case '\x7F': print_to_stream(*stream_, R"(\u007F)"sv); break;
255 case '\t': print_to_stream(*stream_, real_tabs_allowed ? "\t"sv : R"(\t)"sv); break;
256 case '\n': print_to_stream(*stream_, multi_line ? "\n"sv : R"(\n)"sv); break;
257 default:
258 {
259 // control characters from lookup table
260 if TOML_UNLIKELY(c >= '\x00' && c <= '\x1F')
261 print_to_stream(*stream_, control_char_escapes[c]);
262
263 // regular characters
264 else
265 print_to_stream(*stream_, c);
266 }
267 }
268 }
269 }
270
271 // unicode slow path
272 else
273 {
274 utf8_decoder decoder;
275 const char* cp_start = str.data();
276 const char* cp_end = cp_start;
277 for (auto c : str)
278 {
279 decoder(c);
280 cp_end++;
281
282 // if the decoder encounters malformed unicode just emit raw bytes and
283 if (decoder.error())
284 {
285 while (cp_start != cp_end)
286 {
287 print_to_stream(*stream_, R"(\u00)"sv);
288 print_to_stream(*stream_,
289 static_cast<uint8_t>(*cp_start),
290 value_flags::format_as_hexadecimal,
291 2);
292 cp_start++;
293 }
294 decoder.reset();
295 continue;
296 }
297
298 if (!decoder.has_code_point())
299 continue;
300
301 switch (decoder.codepoint)
302 {
303 case U'"': print_to_stream(*stream_, R"(\")"sv); break;
304 case U'\\': print_to_stream(*stream_, R"(\\)"sv); break;
305 case U'\x7F': print_to_stream(*stream_, R"(\u007F)"sv); break;
306 case U'\t': print_to_stream(*stream_, real_tabs_allowed ? "\t"sv : R"(\t)"sv); break;
307 case U'\n': print_to_stream(*stream_, multi_line ? "\n"sv : R"(\n)"sv); break;
308 default:
309 {
310 // control characters from lookup table
311 if TOML_UNLIKELY(decoder.codepoint <= U'\x1F')
312 print_to_stream(*stream_,
313 control_char_escapes[static_cast<uint_least32_t>(decoder.codepoint)]);
314
315 // escaped unicode characters
316 else if (decoder.codepoint > U'\x7F'
317 && (!unicode_allowed || is_non_ascii_vertical_whitespace(decoder.codepoint)))
318 {
319 if (static_cast<uint_least32_t>(decoder.codepoint) > 0xFFFFu)
320 {
321 print_to_stream(*stream_, R"(\U)"sv);
322 print_to_stream(*stream_,
323 static_cast<uint_least32_t>(decoder.codepoint),
324 value_flags::format_as_hexadecimal,
325 8);
326 }
327 else
328 {
329 print_to_stream(*stream_, R"(\u)"sv);
330 print_to_stream(*stream_,
331 static_cast<uint_least32_t>(decoder.codepoint),
332 value_flags::format_as_hexadecimal,
333 4);
334 }
335 }
336
337 // regular characters
338 else
339 print_to_stream(*stream_, cp_start, static_cast<size_t>(cp_end - cp_start));
340 }
341 }
342
343 cp_start = cp_end;
344 }
345 }
346
347 print_unformatted(multi_line ? R"(""")"sv : R"(")"sv);
348 }

356 TOML_EXTERNAL_LINKAGE
357 void formatter::print(const value<int64_t>& val)
358 {
359 naked_newline_ = false;
360
361 if (*val >= 0 && !!int_format_mask_)
362 {
363 static constexpr auto value_flags_mask =
364 value_flags::format_as_binary | value_flags::format_as_octal | value_flags::format_as_hexadecimal;
365
366 const auto fmt = val.flags() & value_flags_mask;
367 switch (fmt)
368 {
369 case value_flags::format_as_binary:
370 if (!!(int_format_mask_ & format_flags::allow_binary_integers))
371 {
372 print_to_stream(*stream_, "0b"sv);
373 print_to_stream(*stream_, *val, fmt);
374 return;
375 }
376 break;
377
378 case value_flags::format_as_octal:
379 if (!!(int_format_mask_ & format_flags::allow_octal_integers))
380 {
381 print_to_stream(*stream_, "0o"sv);
382 print_to_stream(*stream_, *val, fmt);
383 return;
384 }
385 break;
386
387 case value_flags::format_as_hexadecimal:
388 if (!!(int_format_mask_ & format_flags::allow_hexadecimal_integers))
389 {
390 print_to_stream(*stream_, "0x"sv);
391 print_to_stream(*stream_, *val, fmt);
392 return;
393 }
394 break;
395
396 default: break;
397 }
398 }
399
400 // fallback to decimal
401 print_to_stream(*stream_, *val);
402 }

404 TOML_EXTERNAL_LINKAGE
405 void formatter::print(const value<double>& val)
406 {
407 const std::string_view* inf_nan = nullptr;
408 switch (fpclassify(*val))
409 {
410 case fp_class::neg_inf: inf_nan = &constants_->float_neg_inf; break;
411 case fp_class::pos_inf: inf_nan = &constants_->float_pos_inf; break;
412 case fp_class::nan: inf_nan = &constants_->float_nan; break;
413 case fp_class::ok:
414 print_to_stream(*stream_,
415 *val,
416 value_flags::none,
417 !!(config_.flags & format_flags::relaxed_float_precision));
418 break;
419 default: TOML_UNREACHABLE;
420 }
421
422 if (inf_nan)
423 {
424 if (!!(config_.flags & format_flags::quote_infinities_and_nans))
425 print_to_stream_bookended(*stream_, *inf_nan, '"');
426 else
427 print_to_stream(*stream_, *inf_nan);
428 }
429
430 naked_newline_ = false;
431 }
